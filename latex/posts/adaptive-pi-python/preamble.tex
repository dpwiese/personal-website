\def\psfragfigcopy{\begingroup\catcode92 12 \execB}
\def\execB#1#2%
{
\immediate\write18{cp #1#2.eps #2.eps}%
% A dirty hack to fix: Package pstool Warning: No file "*.tex" or "*-psfrag.tex" can be found that may contain macros for "*.eps"
\immediate\write18{touch #2.tex}%
\endgroup
}

\def\psfragfigcleanup{\begingroup\catcode92 12 \execA}
\def\execA#1%
{
\immediate\write18{rm #1.eps}
\immediate\write18{rm #1.pdf}
\immediate\write18{rm #1-pstool.aux}
\immediate\write18{rm #1.tex}
\endgroup
}
